32 /**
33  * @file robotlocalization.cpp
34  * @brief A localization system for a mobile robot with Lag-based feedback loop.
35  *
36  * @section Introduction
37  * Back in the 2000s all [_citation needed_] robots were _differential wheeled
38  * robots_ [1] and localization with sensor fusion was a popular problem.
39  * This example shows a simple solution to that problem using DetectorGraph.
40  * It uses an Extended Kalman Filter [2] to continuously predict and correct
41  * the robot's Pose (and associated uncertainty) from two information sources:
42  * the input to the wheels and a GPS-like system.
43  *
44  * @section fpvslag FuturePublisher vs. Lag<T>
45  * This localization algorithm depends on a feedback-loop in that the output of
46  * one graph evaluation (i.e. `LocalizationBelief`) is used as an input for the
47  * next one (i.e. `Lagged<LocalizationBelief>`):
48  @snippet robotlocalization.cpp KalmanPoseCorrector Feedback Loop
49  * Note that in this graph the TopicState where the feedback loop closes is a
50  * legitimate output in itself and it's very likely that new Detectors in the
51  * graph would subscribe to it. In cases like this the use of
52  * DetectorGraph::Lag is preferred as it is more extensible and it preserves
53  * the normal TopicState guarantee for the TopicState in question as well as it
54  * makes the Lagged version of the TopicState clearly documented and also
55  * available. Lag allows even for detectors that subscribes to both the immediate
56  * and the Lagged version of a TopicState - this could be useful for things like
57  * differentiation etc.
58  *
59  * @section ssc Configuration TopicStates
60  * It also shows how TopicStates can be used for static/configuration data
61  * (e.g. `RobotConfig`). This allows for easy dependency tracking, visualization
62  * and testing at pretty much no runtime cost.
63  *
64  * @section la Localization Algorithm
65  * Regarding the Localization algorithm itself, it's mostly taken from [3]
66  * except for the correction model where this example uses a different input
67  * (i.e. `GPSPosition`) with much simpler transfer function to the Pose vector.
68  *
69  * @section Architecture
70  * The graph below shows the relationships between the topics (rectangles) and
71  * detectors (ellipses). Note that this graph can be automatically generated
72  * for any instance of DetectorGraph::Graph using DetectorGraph::GraphAnalyzer.

206 class ExtendedKalmanPosePredictor : public Detector,
207  public SubscriberInterface<RobotConfig>,
208  public SubscriberInterface<InitialPose>,
209  public SubscriberInterface< Lagged<LocalizationBelief> >,
210  public SubscriberInterface<WheelSpeeds>,
211  public Publisher<LocalizationBelief>
212 {
213 public:
214  ExtendedKalmanPosePredictor(Graph* graph) : Detector(graph), mCurrentBelief(), mConfig()
215  {
216  Subscribe<RobotConfig>(this);
217  Subscribe<InitialPose>(this);
218  Subscribe< Lagged<LocalizationBelief> >(this);
219  Subscribe<WheelSpeeds>(this);
220  SetupPublishing<LocalizationBelief>(this);
221  }

223  virtual void Evaluate(const RobotConfig& aConfig)
224  {
225  mConfig = aConfig;
226  }
227 
228  virtual void Evaluate(const InitialPose& aInitialPose)
229  {
230  mCurrentBelief.state = KalmanState(aInitialPose.pose, Matrix3d::Zero());
231  mCurrentBelief.timestampMs = aInitialPose.timestampMs;
232  }
233 
234  virtual void Evaluate(const Lagged<LocalizationBelief>& aLaggedBelief)
235  {
236  mCurrentBelief.state = aLaggedBelief.data.state;
237  mCurrentBelief.timestampMs = aLaggedBelief.data.timestampMs;
238  }
239 
240  virtual void Evaluate(const WheelSpeeds& aWheelSpeeds)
241  {
242  // [\Delta S_{r}, \Delta S_{l}] from phi
243  Vector2d wheelLinearSpeed = aWheelSpeeds.phi * mConfig.r;
244 
245  // Eq. 2.36
246  double d_th = (wheelLinearSpeed.x() - wheelLinearSpeed.y()) / mConfig.b;
247  // Eq. 2.37
248  double d_travel = (wheelLinearSpeed.x() + wheelLinearSpeed.y()) / 2.;
249 
250  double theta = mCurrentBelief.state.pose.z();
251 
252  // Used on most equations below
253  double th_dth2 = theta + d_th / 2.;
254 
255  // Eq. 2.33 on [3]
256  Vector3d cartesianSpeed;
257  cartesianSpeed << d_travel * cos(th_dth2),
258  d_travel * sin(th_dth2),
259  d_th;
260 
261 
262  // Compute Time Delta
263  double tDelta = (aWheelSpeeds.timestampMs - mCurrentBelief.timestampMs) * 1e-3;
264 
265  // Eq. 2.33 (i.e. 2.23) on [3]
266  mCurrentBelief.state.pose = mCurrentBelief.state.pose + (cartesianSpeed * tDelta);
267  mCurrentBelief.state.pose.z() = WrapAngle(mCurrentBelief.state.pose.z());
268 
269  // Eq. 2.34 on [3]
270  Matrix3d G;
271  G << 1, 0, -d_travel * sin(th_dth2),
272  0, 1, d_travel * cos(th_dth2),
273  0, 0, 1;
274 
275  // Eq. 2.35 on [3]
276  Matrix<double, 3, 2> V;
277  V << 0.5 * cos(th_dth2) - (d_travel / (2. * mConfig.b)) * sin(th_dth2), 0.5 * cos(th_dth2) + (d_travel / (2. * mConfig.b)) * sin(th_dth2),
278  0.5 * sin(th_dth2) + (d_travel / (2. * mConfig.b)) * cos(th_dth2), 0.5 * sin(th_dth2) - (d_travel / (2. * mConfig.b)) * cos(th_dth2),
279  1./mConfig.b, -1./mConfig.b;
280 
281  // Eq. 2.38 on [3]
282  Matrix2d M;
283  M << mConfig.e * abs(wheelLinearSpeed.x()), 0,
284  0, mConfig.e * abs(wheelLinearSpeed.y());
285 
286  // Eq. 2.24 on [3]
287  mCurrentBelief.state.error = G * mCurrentBelief.state.error * G.transpose() + V * M * V.transpose();
288 
289  mCurrentBelief.timestampMs = aWheelSpeeds.timestampMs;
290  Publish(mCurrentBelief);
291  }
292 
293  LocalizationBelief mCurrentBelief;
294  RobotConfig mConfig;
295 };

297 ///! [KalmanPoseCorrector Feedback Loop]
298 class KalmanPoseCorrector : public Detector,
299  public SubscriberInterface< Lagged<LocalizationBelief> >,
300  public SubscriberInterface<GPSPosition>,
301  public Publisher<LocalizationBelief>
302 {
303 public:
304  KalmanPoseCorrector(Graph* graph) : Detector(graph), mCurrentBelief()
305  {
306  Subscribe< Lagged<LocalizationBelief> >(this);
307  Subscribe<GPSPosition>(this);
308  SetupPublishing<LocalizationBelief>(this);
309  }
310 ///! [KalmanPoseCorrector Feedback Loop]
311 
312  virtual void Evaluate(const Lagged<LocalizationBelief>& aLaggedBelief)
313  {
314  mCurrentBelief.state = aLaggedBelief.data.state;
315  mCurrentBelief.timestampMs = aLaggedBelief.data.timestampMs;
316  }
317 
318  virtual void Evaluate(const GPSPosition& aCorrectionInput)
319  {
320  // C = Identity(3)
321 
322  // Eq. 2.15 on [3]
323  Matrix3d K = mCurrentBelief.state.error * (mCurrentBelief.state.error + aCorrectionInput.state.error).inverse();
324 
325  // Eq. 2.13 on [3]
326  mCurrentBelief.state.pose = mCurrentBelief.state.pose + K * (aCorrectionInput.state.pose - mCurrentBelief.state.pose);
327 
328  // Eq. 2.14 on [3]
329  mCurrentBelief.state.error = (Matrix3d::Identity() - K) * mCurrentBelief.state.error;
330 
331  Publish(mCurrentBelief);
332  }
333 
334  LocalizationBelief mCurrentBelief;
335 };
